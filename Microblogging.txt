  <h2>Share Your Thoughts</h2>

  <div class="post-form">
    <input type="text" id="username" placeholder="Enter your name" required>
    <textarea id="content" placeholder="What's on your mind?"></textarea>
    <input type="file" id="imageInput" accept="image/*">
    <button onclick="addPost()">Post</button>
  </div>

  <div id="stats"></div>
  <div id="posts"></div>

<script>
  // Default post by Thiru
  let posts = [
    {
      username: "Thiru",
      content: "Everyday is a good day with positive thoughts.",
      imageUrl: null,
      likes: 1200,
      shares: 250,
      comments: [],
      time: new Date().toLocaleString()
    }
  ];

  function addPost() {
    const username = document.getElementById('username').value.trim();
    const content = document.getElementById('content').value.trim();
    const imageFile = document.getElementById('imageInput').files[0];

    if (!username || !content) {
      alert("Please enter your name and content.");
      return;
    }

    const reader = new FileReader();
    reader.onload = function(event) {
      const imageUrl = imageFile ? event.target.result : null;
      const post = {
        username,
        content,
        imageUrl,
        likes: 0,
        shares: 0,
        comments: [],
        time: new Date().toLocaleString()
      };
      posts.unshift(post);
      document.getElementById('content').value = '';
      document.getElementById('imageInput').value = '';
      renderPosts();
    };

    if (imageFile) reader.readAsDataURL(imageFile);
    else {
      const post = {
        username,
        content,
        imageUrl: null,
        likes: 0,
        shares: 0,
        comments: [],
        time: new Date().toLocaleString()
      };
      posts.unshift(post);
      document.getElementById('content').value = '';
      renderPosts();
    }
  }

  function likePost(index) {
    posts[index].likes++;
    renderPosts();
  }

  function sharePost(index) {
    posts[index].shares++;
    alert("Post shared successfully!");
    renderPosts();
  }

  function addComment(index) {
    const input = document.getElementById(`commentInput-${index}`);
    const text = input.value.trim();
    if (text) {
      posts[index].comments.push(text);
      input.value = '';
      renderPosts();
    }
  }

  function formatCount(n) {
    if (n >= 1000) return (n / 1000).toFixed(1) + "k";
    return n;
  }

  function renderPosts() {
    const container = document.getElementById('posts');
    container.innerHTML = '';
    document.getElementById('stats').innerHTML = `<b>Total Users:</b> ${posts.length}`;

    posts.forEach((post, index) => {
      const div = document.createElement('div');
      div.className = 'post';
      div.innerHTML = `
        <div class="post-header">
          <span><b>${post.username}</b></span>
          <span>${post.time}</span>
        </div>
        <div class="post-content">${post.content}</div>
        ${post.imageUrl ? `<img src="${post.imageUrl}" alt="post image">` : ''}
        <div class="actions">
          <button onclick="likePost(${index})">‚ù§Ô∏è Like (${formatCount(post.likes)})</button>
          <button onclick="sharePost(${index})">üîÅ Share (${formatCount(post.shares)})</button>
        </div>
        <div class="comments">
          <input type="text" id="commentInput-${index}" placeholder="Add a comment...">
          <button onclick="addComment(${index})">Post</button>
          ${post.comments.map(c => `<div class='comment'>${c}</div>`).join('')}
        </div>
      `;
      container.appendChild(div);
    });
  }

  renderPosts();
</script>
